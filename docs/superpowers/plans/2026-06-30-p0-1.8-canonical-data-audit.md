# §1.8 Canonical Data Model + Local Audit Log — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:executing-plans (inline, single-agent). Steps
> use checkbox (`- [ ]`) syntax.

**Goal:** The canonical M1–M3 relational schema + roll-forward/back migrations + a canonical append-only
audit store (dual-timestamp, read+write actions) with non-invasive reconciliation of the §1.5/§1.7
forerunners, + the entity↔memory mapping doc. Net-new; standalone (no `core`/`agent_loop` change).

**Architecture:** New `data/` package over SQLite. `migrations` owns the schema; `CanonicalStore` runs it +
exposes CRUD + `.audit`; `AuditStore` is append-only over the same connection and can import the §1.5
governance audit + §1.7 transitions. Spec:
`docs/superpowers/specs/2026-06-30-p0-1.8-canonical-data-audit-design.md`.

**Tech Stack:** Python ≥3.11, stdlib only (`sqlite3`, `json`, `time`, `datetime`, `dataclasses`), pytest.

# data/migrations.py
from __future__ import annotations
import sqlite3
from dataclasses import dataclass
from .schema import TABLES

@dataclass
class Migration:
    version: int
    up: str
    down: str

# v1: the full M1-M3 subset + audit_log (one migration; future entities add v2, v3, ...).
_V1_UP = ";\n".join(TABLES.values())
_V1_DOWN = ";\n".join(f"DROP TABLE IF EXISTS {name}" for name in TABLES)
MIGRATIONS = [Migration(1, _V1_UP, _V1_DOWN)]
LATEST = max(m.version for m in MIGRATIONS)

def current_version(conn: sqlite3.Connection) -> int:
    conn.execute("CREATE TABLE IF NOT EXISTS schema_version (version INTEGER)")
    row = conn.execute("SELECT version FROM schema_version").fetchone()
    return row[0] if row else 0

def _set_version(conn, v):
    conn.execute("DELETE FROM schema_version")
    conn.execute("INSERT INTO schema_version (version) VALUES (?)", (v,))

def migrate(conn: sqlite3.Connection, to_version: int = LATEST) -> None:
    v = current_version(conn)
    if to_version > v:
        for m in sorted(MIGRATIONS, key=lambda m: m.version):
            if v < m.version <= to_version:
                conn.executescript(m.up); v = m.version
    elif to_version < v:
        for m in sorted(MIGRATIONS, key=lambda m: m.version, reverse=True):
            if to_version < m.version <= v:
                conn.executescript(m.down); v = m.version - 1
    _set_version(conn, to_version)
    conn.commit()
```

# data/audit.py
from __future__ import annotations
import sqlite3, time
from dataclasses import dataclass
from datetime import datetime, timezone
from typing import List, Optional

@dataclass
class AuditRecord:
    actor: str; action: str; target_key: str; at_monotonic: float; at_wall: str; reason: str; result: str

class AuditStore:
    def __init__(self, conn: sqlite3.Connection) -> None:
        self._conn = conn  # the migrated canonical connection (audit_log exists)

    def _insert(self, r: AuditRecord) -> None:
        self._conn.execute("INSERT INTO audit_log (actor, action, target_key, at_monotonic, at_wall, reason, result) "
                           "VALUES (?, ?, ?, ?, ?, ?, ?)",
                           (r.actor, r.action, r.target_key, r.at_monotonic, r.at_wall, r.reason, r.result))
        self._conn.commit()

    def record(self, actor, action, target_key, *, reason="", result="ok") -> None:
        self._insert(AuditRecord(actor, action, target_key, time.monotonic(),
                                 datetime.now(timezone.utc).isoformat(), reason, result))

    def query(self, *, target_key=None, actor=None, action=None, result_prefix=None) -> List[AuditRecord]:
        sql = "SELECT actor, action, target_key, at_monotonic, at_wall, reason, result FROM audit_log"
        clauses, params = [], []
        if target_key is not None: clauses.append("target_key=?"); params.append(target_key)
        if actor is not None: clauses.append("actor=?"); params.append(actor)
        if action is not None: clauses.append("action=?"); params.append(action)
        if result_prefix is not None: clauses.append("result LIKE ?"); params.append(result_prefix + "%")
        if clauses: sql += " WHERE " + " AND ".join(clauses)
        sql += " ORDER BY id"
        return [AuditRecord(*row) for row in self._conn.execute(sql, params).fetchall()]

    def import_governance_audit(self, governance_audit_log) -> int:
        rows = governance_audit_log.query()  # §1.5 AuditLog -> same-shaped AuditRecords
        for r in rows:
            self._insert(AuditRecord(r.actor, r.action, r.target_key, r.at_monotonic, r.at_wall, r.reason, r.result))
        return len(rows)

    def import_transitions(self, transitions) -> int:
        n = 0
        for t in transitions:  # duck-typed §1.7 Transition
            self._insert(AuditRecord(t.actor, "transition", t.instance_id, 0.0, t.at,
                                     f"{t.from_state}->{t.to_state}:{t.trigger}", "ok"))
            n += 1
        return n
```
