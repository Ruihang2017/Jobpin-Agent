# §1.5 HR Memory Governance — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:executing-plans (inline, single-agent —
> the repo convention). Steps use checkbox (`- [ ]`) syntax.

**Goal:** Wrap every memory write/recall in a governance shell (namespace + provenance + consent +
retention + RBAC + audit + bias hygiene) and add the governed model-facing `memory` write tool —
making the §1.4 seams real — with no change to `agent_loop.py` and the ported `MemoryStore` unchanged.

**Architecture:** A new `src/jobpin_agent/governance/` package consumed by the existing memory paths.
Governance is enforced in the **governed tool's write path** (the sole writer to the curated store) and
on the **recall path** via the existing `scope_filter` / `search(scope=)` filter-before-NN seam.
Erasure orchestrates the §1.4 `delete_by_key_prefix` cascade. Audit is an append-only SQLite log.

**Tech Stack:** Python ≥3.11, stdlib + `sqlite3` (no new deps), pytest. Spec:
`docs/superpowers/specs/2026-06-30-p0-1.5-hr-memory-governance-design.md`.

```python
# agent/src/jobpin_agent/governance/namespace.py
from __future__ import annotations
from dataclasses import dataclass

ENTITY_TYPES = frozenset({"candidate", "employee", "job", "org", "recruiter", "semantic"})
DEFAULT_TENANT = "acme"
DEFAULT_ORG = "apac"
_LEVELS = {"tenant": 1, "org": 2, "entity_type": 3}

@dataclass(frozen=True)
class MemoryKey:
    tenant: str
    org: str
    entity_type: str
    entity_id: str
    def format(self) -> str:
        return f"{self.tenant}:{self.org}:{self.entity_type}:{self.entity_id}"
    def prefix(self, level: str) -> str:
        parts = [self.tenant, self.org, self.entity_type, self.entity_id]
        return ":".join(parts[:_LEVELS[level]])

def parse(key: str) -> MemoryKey:
    parts = key.split(":")
    if len(parts) != 4 or not all(parts):
        raise ValueError(f"malformed memory_key: {key!r} (want tenant:org:entity_type:entity_id)")
    if parts[2] not in ENTITY_TYPES:
        raise ValueError(f"unknown entity_type in {key!r}: {parts[2]}")
    return MemoryKey(*parts)

def is_valid(key: str) -> bool:
    try:
        parse(key)
        return True
    except ValueError:
        return False
```

```python
# agent/src/jobpin_agent/governance/labels.py
from __future__ import annotations
from dataclasses import dataclass

CONSENT_REQUIRED_SOURCE_TYPES = frozenset({"candidate_submitted", "candidate_volunteered", "third_party"})
LEGAL_BASES = frozenset({"consent", "legitimate_interest", "contract"})
_HEADER_SEP = "---"

@dataclass
class Provenance:
    memory_key: str
    source_type: str
    source_ref: str = ""
    collected_at: str = ""
    collected_by: str = ""

@dataclass
class ConsentLabel:
    legal_basis: str
    purpose: str = ""
    consent_id: str = ""
    use_scope: str = ""

@dataclass
class RetentionPolicy:
    policy_key: str
    ttl_days: int
    basis: str = ""

def render_header(prov: Provenance, consent: ConsentLabel, retention_key: str) -> str:
    lines = [
        f"key: {prov.memory_key}",
        f"source_type: {prov.source_type}",
        f"source_ref: {prov.source_ref}",
        f"collected_at: {prov.collected_at}",
        f"collected_by: {prov.collected_by}",
        f"legal_basis: {consent.legal_basis}",
        f"consent_id: {consent.consent_id}",
        f"purpose: {consent.purpose}",
        f"retention_ttl: {retention_key}",
        _HEADER_SEP,
        "",
    ]
    return "\n".join(lines)

def parse_header(entry: str):
    if f"\n{_HEADER_SEP}\n" not in entry and not entry.startswith(f"{_HEADER_SEP}\n"):
        if f"\n{_HEADER_SEP}\n" not in entry:
            return None, entry
    head, _, body = entry.partition(f"\n{_HEADER_SEP}\n")
    if not _:
        return None, entry
    labels = {}
    for line in head.splitlines():
        if ":" in line:
            k, _sep, v = line.partition(":")
            labels[k.strip()] = v.strip()
    return labels, body
```

```python
# agent/src/jobpin_agent/governance/audit.py
from __future__ import annotations
import sqlite3, time
from dataclasses import dataclass
from datetime import datetime, timezone
from pathlib import Path
from typing import List, Optional

@dataclass
class AuditRecord:
    actor: str
    action: str
    target_key: str
    at_monotonic: float
    at_wall: str
    reason: str
    result: str

class AuditLog:
    def __init__(self, db_path: str = ":memory:") -> None:
        if db_path != ":memory:":
            Path(db_path).parent.mkdir(parents=True, exist_ok=True)
        self._conn = sqlite3.connect(db_path)
        self._conn.execute(
            "CREATE TABLE IF NOT EXISTS audit_log ("
            "id INTEGER PRIMARY KEY AUTOINCREMENT, actor TEXT, action TEXT, target_key TEXT, "
            "at_monotonic REAL, at_wall TEXT, reason TEXT, result TEXT)")
        self._conn.commit()

    def record(self, actor: str, action: str, target_key: str, *, reason: str = "", result: str = "ok") -> None:
        self._conn.execute(
            "INSERT INTO audit_log (actor, action, target_key, at_monotonic, at_wall, reason, result) "
            "VALUES (?, ?, ?, ?, ?, ?, ?)",
            (actor, action, target_key, time.monotonic(),
             datetime.now(timezone.utc).isoformat(), reason, result))
        self._conn.commit()

    def query(self, *, target_key: Optional[str] = None, action: Optional[str] = None) -> List[AuditRecord]:
        sql = "SELECT actor, action, target_key, at_monotonic, at_wall, reason, result FROM audit_log"
        clauses, params = [], []
        if target_key is not None:
            clauses.append("target_key = ?"); params.append(target_key)
        if action is not None:
            clauses.append("action = ?"); params.append(action)
        if clauses:
            sql += " WHERE " + " AND ".join(clauses)
        sql += " ORDER BY id"
        return [AuditRecord(*row) for row in self._conn.execute(sql, params).fetchall()]
```

```python
# agent/src/jobpin_agent/governance/bias_hygiene.py
from __future__ import annotations
import re
from dataclasses import dataclass
from typing import Optional, Tuple

@dataclass
class BiasFinding:
    code: str
    term: str
    reason: str

# Protected attributes (Australian anti-discrimination grounds) — using these as decision features is rejected.
PROTECTED_ATTRIBUTES: Tuple[str, ...] = (
    "age", "years old", "gender", "male", "female", "race", "ethnic", "nationality",
    "married", "marital", "pregnan", "children", "family status", "religion", "disability",
    "health condition", "sexual orientation",
)
# Proxy variables / indirect-discrimination patterns — flagged (block when used as a hard threshold).
PROXY_PATTERNS: Tuple[Tuple[str, str], ...] = (
    (r"\bgraduat(e|ed)\b.{0,40}\b(group of eight|go8|ivy league|sandstone|elite)\b",
     "elite-school requirement is a socio-economic proxy"),
    (r"\b(must|required to).{0,30}\b(postcode|suburb)\b", "postcode is a proxy for race/SES"),
    (r"\bgraduation year\b", "graduation year is an age proxy"),
    (r"\bnative (english )?speaker\b", "native-speaker requirement is a national-origin proxy"),
)

def scan(text: str) -> Optional[BiasFinding]:
    low = text.lower()
    for term in PROTECTED_ATTRIBUTES:
        if term in low:
            return BiasFinding("rejected:bias", term, f"protected attribute referenced: '{term}'")
    for pattern, reason in PROXY_PATTERNS:
        if re.search(pattern, low):
            return BiasFinding("flagged:bias", pattern, reason)
    return None
```

```python
from __future__ import annotations
from dataclasses import dataclass
from typing import Callable, Tuple

@dataclass(frozen=True)
class Principal:
    user_id: str
    role: str
    allowed_scopes: Tuple[str, ...]

FULL_ACCESS = Principal(user_id="system", role="system", allowed_scopes=("",))

def scope_predicate(principal: Principal) -> Callable[[str], bool]:
    scopes = principal.allowed_scopes
    def allow(memory_key: str) -> bool:
        for s in scopes:
            if s == "" or memory_key == s or memory_key.startswith(s + ":"):
                return True
        return False
    return allow
```

```python
# agent/src/jobpin_agent/governance/retention.py
from __future__ import annotations
from typing import List, Tuple
from .labels import RetentionPolicy

RETENTION_POLICIES = {
    "hired_5y": RetentionPolicy("hired_5y", 365 * 5, "employment record retention"),
    "not_hired_180d": RetentionPolicy("not_hired_180d", 180, "unsuccessful applicant retention"),
    "withdrawn_30d": RetentionPolicy("withdrawn_30d", 30, "withdrawn applicant minimisation"),
}

def sweep(now_days: float, items: List[Tuple[str, float, str]]) -> List[str]:
    expired = []
    for memory_key, collected_at_days, policy_key in items:
        policy = RETENTION_POLICIES.get(policy_key)
        if policy and (now_days - collected_at_days) > policy.ttl_days:
            expired.append(memory_key)
    return expired

class BackupAgeingRegister:
    def __init__(self) -> None:
        self._entries: List[Tuple[str, float, float]] = []
    def register(self, memory_key: str, erased_at_days: float, ages_out_at_days: float) -> None:
        self._entries.append((memory_key, erased_at_days, ages_out_at_days))
    def pending(self, now_days: float) -> List[str]:
        return [mk for mk, _erased, ages_out in self._entries if now_days < ages_out]
```

```python
# agent/src/jobpin_agent/governance/write_gate.py
from __future__ import annotations
from dataclasses import dataclass
from .audit import AuditLog
from .bias_hygiene import scan as default_bias_scan
from .labels import (Provenance, ConsentLabel, render_header, CONSENT_REQUIRED_SOURCE_TYPES)

@dataclass
class Decision:
    ok: bool
    code: str = ""
    header: str = ""

class GovernanceGate:
    def __init__(self, audit: AuditLog, bias_scan=default_bias_scan) -> None:
        self.audit = audit
        self._bias_scan = bias_scan

    def validate(self, action: str, target_key: str, body: str,
                 prov: Provenance, consent: ConsentLabel, retention_key: str, *, actor: str) -> Decision:
        if action == "remove":
            return Decision(ok=True)
        if not prov.source_ref:
            self.audit.record(actor, f"write:{action}", target_key, result="rejected:no_provenance")
            return Decision(False, "rejected:no_provenance")
        if prov.source_type in CONSENT_REQUIRED_SOURCE_TYPES and not consent.consent_id:
            self.audit.record(actor, f"write:{action}", target_key, result="rejected:no_consent")
            return Decision(False, "rejected:no_consent")
        finding = self._bias_scan(body)
        if finding is not None:
            self.audit.record(actor, f"write:{action}", target_key, reason=finding.reason, result=finding.code)
            return Decision(False, finding.code)
        return Decision(True, header=render_header(prov, consent, retention_key))
```

```python
# builtin.py — constructor
def __init__(self, store, *, gate=None, actor="system"):
    self._store = store
    self._gate = gate
    self._actor = actor

# capture actor from initialize if provided
def initialize(self, session_id, **kwargs):
    ident = kwargs.get("agent_identity")
    if ident:
        self._actor = ident
    return None

_TARGET_KEYS = {"org": f"{DEFAULT_TENANT}:{DEFAULT_ORG}:org:policy",
                "recruiter": f"{DEFAULT_TENANT}:{DEFAULT_ORG}:recruiter:prefs"}

def get_tool_schemas(self):
    if self._gate is None:
        return []
    return [MEMORY_TOOL_SCHEMA]   # the dict above, module-level constant

def handle_tool_call(self, tool_name, args, **kwargs):
    if tool_name != "memory" or self._gate is None:
        return super().handle_tool_call(tool_name, args, **kwargs)
    action = str(args.get("action") or "")
    target = str(args.get("target") or "org")
    if target not in ("org", "recruiter"):
        return json.dumps({"success": False, "error": f"unknown target {target!r}"})
    body = str(args.get("content") or "")
    key = self._TARGET_KEYS[target]
    prov = Provenance(key, str(args.get("source_type") or ""), str(args.get("source_ref") or ""),
                      collected_by=self._actor)
    consent = ConsentLabel(str(args.get("legal_basis") or "legitimate_interest"),
                           str(args.get("purpose") or ""), str(args.get("consent_id") or ""))
    retention_key = str(args.get("retention_policy") or "not_hired_180d")
    decision = self._gate.validate(action, key, body, prov, consent, retention_key, actor=self._actor)
    if not decision.ok:
        return json.dumps({"success": False, "rejected": decision.code.split(":", 1)[-1],
                           "code": decision.code})
    if action == "remove":
        result = self._store.remove(target, str(args.get("old_text") or ""))
    elif action == "replace":
        result = self._store.replace(target, str(args.get("old_text") or ""), decision.header + body)
    else:
        result = self._store.add(target, decision.header + body)
    if result.get("success"):
        self._gate.audit.record(self._actor, f"write:{action}", key, result="ok")
    return json.dumps(result)
```

```python
# agent/src/jobpin_agent/governance/erasure.py
from __future__ import annotations
from typing import Callable, Dict, List
from .audit import AuditLog
from .retention import BackupAgeingRegister

class Eraser:
    def __init__(self, audit: AuditLog, register: BackupAgeingRegister,
                 cache_clearers: List[Callable[[], None]]) -> None:
        self._audit = audit
        self._register = register
        self._clearers = list(cache_clearers)

    def erase(self, memory_key: str, *, actor: str, reason: str = "",
              deleter: Callable[[str], Dict[str, int]],
              now_days: float = 0.0, ages_out_at_days: float = 180.0) -> dict:
        deleted = deleter(memory_key)
        for clear in self._clearers:
            try:
                clear()
            except Exception:
                pass
        self._audit.record(actor, "erase", memory_key,
                           reason=f"{reason} | deleted={deleted}", result="ok")
        self._register.register(memory_key, erased_at_days=now_days, ages_out_at_days=ages_out_at_days)
        return {"deleted": deleted, "audit": "ok"}
```
