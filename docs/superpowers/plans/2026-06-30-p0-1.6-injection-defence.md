# §1.6 Injection-defence + pre-compression — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:executing-plans (inline, single-agent). Steps
> use checkbox (`- [ ]`) syntax.

**Goal:** Port Hermes's context-window security (threat patterns + streaming scrubber) to local, force a
scan+fence on external text, and wire pre-compression fact-injection (capture `on_pre_compress` → merge
into the summary + gated persist) — fixing the two Hermes gaps.

**Architecture:** New `security/` package + `core/compression.py`; a minimal opt-in `agent_loop.py`
compression call; real scan wired into the §1.2/§1.4 `scan_entry` seams. Spec:
`docs/superpowers/specs/2026-06-30-p0-1.6-injection-defence-design.md`.

**Tech Stack:** Python ≥3.11, stdlib only (`re`, `sqlite3`), pytest.

from __future__ import annotations
from dataclasses import dataclass, field
from typing import Callable, List
from .threat_patterns import scan_for_threats
from ..memory.fence import build_memory_context_block

@dataclass
class IngestResult:
    ok: bool
    fenced: str = ""
    findings: List[str] = field(default_factory=list)
    blocked: bool = False

def ingest_external_text(text, *, source, scope="context",
                         scan: Callable[[str, str], list] = scan_for_threats,
                         fence: Callable[[str], str] = build_memory_context_block) -> IngestResult:
    findings = scan(text, scope)
    if findings:
        return IngestResult(ok=False, blocked=True, findings=list(findings))
    return IngestResult(ok=True, fenced=fence(text))
```

# session_store.py
def compact(self, session_id: str, summary_message: Message, keep_recent: int) -> None:
    msgs = self.get_messages(session_id)
    if len(msgs) <= keep_recent:
        return
    recent = msgs[-keep_recent:] if keep_recent > 0 else []
    new_msgs = [summary_message, *recent]
    self._conn.execute("DELETE FROM messages WHERE session_id=?", (session_id,))
    for i, m in enumerate(new_msgs):
        self._conn.execute("INSERT INTO messages (session_id, seq, payload) VALUES (?, ?, ?)",
                           (session_id, i, json.dumps(_msg_to_dict(m))))
    self._conn.commit()
```
```python
# core/compression.py
from __future__ import annotations
from dataclasses import dataclass
from typing import Callable, List, Optional
from .messages import Message, Role

def default_summarize(old_messages: List[Message], facts: str) -> str:
    lines = ["[compressed-summary] earlier conversation folded to save context."]
    if facts and facts.strip():
        lines.append("Key facts retained:\n" + facts.strip())
    digest = [f"- {m.role.value}: {m.content.strip()}" for m in old_messages if m.content and m.content.strip()]
    if digest:
        lines.append("Digest of folded turns:\n" + "\n".join(digest))
    return "\n\n".join(lines)

@dataclass
class CompressionResult:
    compressed: bool
    summary: str = ""
    facts: str = ""
    persisted: bool = False

class ContextCompressor:
    def __init__(self, *, max_messages=20, keep_recent=6, summarize_fn=default_summarize, persist_fn=None):
        self._max = max_messages; self._keep = keep_recent
        self._summarize = summarize_fn; self._persist = persist_fn

    def should_compress(self, messages) -> bool:
        return len(messages) > self._max

    def compress(self, session_id, store, hooks) -> CompressionResult:
        msgs = store.get_messages(session_id)
        if len(msgs) <= self._keep:
            return CompressionResult(compressed=False)
        old = msgs[:-self._keep] if self._keep > 0 else msgs
        try:
            facts = hooks.on_pre_compress(old) or ""
        except Exception:
            facts = ""
        summary = self._summarize(old, facts)
        store.compact(session_id, Message(Role.SYSTEM, content=summary), self._keep)
        persisted = False
        if self._persist is not None and facts.strip():
            try:
                persisted = bool(self._persist(facts))
            except Exception:
                persisted = False     # gated persist rejection must not crash the turn
        return CompressionResult(compressed=True, summary=summary, facts=facts, persisted=persisted)
```
(Bilingual docstrings throughout.)
