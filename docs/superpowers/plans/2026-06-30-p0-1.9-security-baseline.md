import os
import sqlcipher3


def test_sqlcipher_pragma_key_roundtrip(tmp_path):
    db = str(tmp_path / "enc.db")
    key_hex = "00112233445566778899aabbccddeeff00112233445566778899aabbccddeeff"
    conn = sqlcipher3.connect(db)
    conn.execute(f"PRAGMA key = \"x'{key_hex}'\"")
    conn.execute("CREATE TABLE t (v TEXT)")
    conn.execute("INSERT INTO t VALUES ('SECRETNAME')")
    conn.commit()
    conn.close()

    # Raw bytes on disk must not contain the plaintext.
    raw = open(db, "rb").read()
    assert b"SECRETNAME" not in raw

    # Wrong key cannot read.
    bad = sqlcipher3.connect(db)
    bad.execute("PRAGMA key = \"x'ffffffffffffffffffffffffffffffffffffffffffffffffffffffffffffffff'\"")
    try:
        bad.execute("SELECT v FROM t").fetchall()
        assert False, "wrong key should fail"
    except Exception:
        pass
    bad.close()

    # Right key reads back.
    good = sqlcipher3.connect(db)
    good.execute(f"PRAGMA key = \"x'{key_hex}'\"")
    assert good.execute("SELECT v FROM t").fetchone()[0] == "SECRETNAME"
    good.close()
```

from __future__ import annotations
import os
from cryptography.hazmat.primitives.ciphers.aead import AESGCM
from cryptography.hazmat.primitives.kdf.hkdf import HKDF
from cryptography.hazmat.primitives import hashes

_NONCE = 12

def derive_subkey(master: bytes, label: str) -> bytes:
    return HKDF(algorithm=hashes.SHA256(), length=32, salt=None,
                info=b"jobpin/" + label.encode()).derive(master)

class Cipher:
    def __init__(self, key: bytes) -> None:
        if len(key) != 32:
            raise ValueError("AES-256 key must be 32 bytes")
        self._aes = AESGCM(key)
    def encrypt(self, plaintext: bytes) -> bytes:
        nonce = os.urandom(_NONCE)
        return nonce + self._aes.encrypt(nonce, plaintext, None)
    def decrypt(self, blob: bytes) -> bytes:
        return self._aes.decrypt(blob[:_NONCE], blob[_NONCE:], None)
```

- [ ] **Step 3: Implement** `keystore.py`. Note the DPAPI ctypes care (buffers held alive; `WinError` on failure):
```python
from __future__ import annotations
import os, secrets, subprocess, sys
from abc import ABC, abstractmethod

class KeyStore(ABC):
    @abstractmethod
    def get_or_create_master_key(self) -> bytes: ...

def _new_key() -> bytes:
    return secrets.token_bytes(32)

class DevKeyStore(KeyStore):
    # INSECURE: stores the master key XOR'd with a fixed pad. CI/Linux/tests only.
    _PAD = b"jobpin-dev-pad-not-secure-000000"  # 32 bytes
    def __init__(self, path: str) -> None:
        self._path = path
    def get_or_create_master_key(self) -> bytes:
        if os.path.exists(self._path):
            wrapped = open(self._path, "rb").read()
            return bytes(a ^ b for a, b in zip(wrapped, self._PAD))
        key = _new_key()
        with open(self._path, "wb") as f:
            f.write(bytes(a ^ b for a, b in zip(key, self._PAD)))
        return key

class WindowsDpapiKeyStore(KeyStore):
    def __init__(self, path: str) -> None:
        self._path = path
    def get_or_create_master_key(self) -> bytes:
        if os.path.exists(self._path):
            return _dpapi_unprotect(open(self._path, "rb").read())
        key = _new_key()
        with open(self._path, "wb") as f:
            f.write(_dpapi_protect(key))
        return key

class MacKeychainKeyStore(KeyStore):
    def __init__(self, service: str = "jobpin-agent", account: str = "master-key") -> None:
        self._service, self._account = service, account
    def get_or_create_master_key(self) -> bytes:
        found = subprocess.run(
            ["security", "find-generic-password", "-s", self._service, "-a", self._account, "-w"],
            capture_output=True, text=True)
        if found.returncode == 0 and found.stdout.strip():
            return bytes.fromhex(found.stdout.strip())
        key = _new_key()
        subprocess.run(
            ["security", "add-generic-password", "-s", self._service, "-a", self._account,
             "-w", key.hex(), "-U"], check=True)
        return key

def default_keystore(path: str) -> KeyStore:
    if sys.platform == "win32":
        return WindowsDpapiKeyStore(path)
    if sys.platform == "darwin":
        return MacKeychainKeyStore()
    return DevKeyStore(path)

# --- DPAPI via ctypes (Windows) ---
def _dpapi_protect(data: bytes) -> bytes:
    return _dpapi_call(data, protect=True)

def _dpapi_unprotect(data: bytes) -> bytes:
    return _dpapi_call(data, protect=False)

def _dpapi_call(data: bytes, *, protect: bool) -> bytes:
    import ctypes
    from ctypes import wintypes

    class BLOB(ctypes.Structure):
        _fields_ = [("cbData", wintypes.DWORD), ("pbData", ctypes.POINTER(ctypes.c_char))]

    buf = ctypes.create_string_buffer(data, len(data))
    blob_in = BLOB(len(data), ctypes.cast(buf, ctypes.POINTER(ctypes.c_char)))
    blob_out = BLOB()
    fn = ctypes.windll.crypt32.CryptProtectData if protect else ctypes.windll.crypt32.CryptUnprotectData
    ok = fn(ctypes.byref(blob_in), None, None, None, None, 0, ctypes.byref(blob_out))
    if not ok:
        raise ctypes.WinError()
    try:
        return ctypes.string_at(blob_out.pbData, blob_out.cbData)
    finally:
        ctypes.windll.kernel32.LocalFree(blob_out.pbData)
```

```python
from __future__ import annotations
from dataclasses import dataclass, field
from typing import Dict, FrozenSet, Tuple
from ..governance.rbac import Principal, scope_predicate, FULL_ACCESS
from ..data.schema import User, Org

@dataclass(frozen=True)
class Role:
    name: str
    permissions: FrozenSet[str]
    scope_entity_types: Tuple[str, ...]      # entity types the role may reach; () + sensitivity gate
    sensitivity_ceiling: str                 # "normal" | "sensitive"

_ALL = frozenset({"read", "write", "delete", "export", "configure"})
ROLE_POLICIES: Dict[str, Role] = {
    "system": Role("system", _ALL, (), "sensitive"),
    "admin": Role("admin", _ALL, (), "sensitive"),
    "recruiter": Role("recruiter", frozenset({"read", "write", "export"}),
                      ("candidate", "job", "application"), "normal"),
    "interviewer": Role("interviewer", frozenset({"read"}), ("candidate",), "normal"),
}
_SENS = {"normal": 0, "sensitive": 1}

@dataclass(frozen=True)
class ResourceRef:
    type: str
    org_id: str
    memory_key: str
    sensitivity: str = "normal"

@dataclass(frozen=True)
class Decision:
    allowed: bool
    reason: str

def principal_for(user: User, org: Org) -> Principal:
    role = ROLE_POLICIES.get(user.role)
    if role is None:
        return Principal(user_id=user.user_id, role=user.role, allowed_scopes=())  # deny-all
    if role.name in ("system", "admin"):
        base = (f"{user.tenant_id}:{user.org_id}",) if role.name == "admin" else ("",)
        return Principal(user_id=user.user_id, role=role.name, allowed_scopes=base)
    scopes = tuple(f"{user.tenant_id}:{user.org_id}:{et}" for et in role.scope_entity_types)
    return Principal(user_id=user.user_id, role=role.name, allowed_scopes=scopes)

def authorize(principal: Principal, action: str, resource: ResourceRef) -> Decision:
    role = ROLE_POLICIES.get(principal.role)
    if role is None:
        return Decision(False, "rejected:rbac")
    if action not in role.permissions:
        return Decision(False, "rejected:rbac")
    if _SENS.get(resource.sensitivity, 1) > _SENS.get(role.sensitivity_ceiling, 0):
        return Decision(False, "rejected:rbac")
    if not scope_predicate(principal)(resource.memory_key):
        return Decision(False, "rejected:rbac")
    return Decision(True, "ok")
```
