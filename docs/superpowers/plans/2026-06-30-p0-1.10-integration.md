```python
from __future__ import annotations
from abc import ABC, abstractmethod
from dataclasses import dataclass
from ..data.schema import Application, Candidate, Job

@dataclass(frozen=True)
class ExternalRecord:
    source: str
    kind: str
    raw: dict

class Connector(ABC):
    name: str = "connector"
    @abstractmethod
    def fetch(self, kind: str) -> list[ExternalRecord]: ...

class AntiCorruptionLayer(ABC):
    def translate(self, rec: ExternalRecord):
        if rec.kind == "candidate":
            return self._to_candidate(rec.raw)
        if rec.kind == "job":
            return self._to_job(rec.raw)
        if rec.kind == "application":
            return self._to_application(rec.raw)
        raise ValueError(f"unknown external kind: {rec.kind}")
    @abstractmethod
    def _to_candidate(self, raw: dict) -> Candidate: ...
    @abstractmethod
    def _to_job(self, raw: dict) -> Job: ...
    @abstractmethod
    def _to_application(self, raw: dict) -> Application: ...
```

```python
from __future__ import annotations
from ..sdk import AntiCorruptionLayer, Connector, ExternalRecord
from ...data.schema import Application, Candidate, Job

_FIXTURES = {
    "candidate": [
        {"ext_id": "A-1", "full_name": "Ada Lovelace", "skill_tags": ["Python", "Math"], "yrs": 7, "loc": "Sydney"},
        {"ext_id": "A-2", "full_name": "Grace Hopper", "skill_tags": ["COBOL", "Compilers"], "yrs": 12, "loc": "Remote"},
    ],
    "job": [{"ext_req": "R-9", "req_title": "Senior Engineer", "state": "open"}],
}

class FakeATSConnector(Connector):
    name = "fake-ats"
    def __init__(self, fixtures: dict | None = None) -> None:
        self._fixtures = fixtures if fixtures is not None else _FIXTURES
    def fetch(self, kind: str) -> list[ExternalRecord]:
        return [ExternalRecord(source=self.name, kind=kind, raw=row) for row in self._fixtures.get(kind, [])]

class FakeATSAntiCorruption(AntiCorruptionLayer):
    def _to_candidate(self, raw: dict) -> Candidate:
        return Candidate(candidate_id=raw["ext_id"], name=raw.get("full_name", ""),
                         skills=list(raw.get("skill_tags", [])), years=int(raw.get("yrs", 0)),
                         location=raw.get("loc", ""))
    def _to_job(self, raw: dict) -> Job:
        return Job(job_id=raw["ext_req"], title=raw.get("req_title", ""), status=raw.get("state", "open"))
    def _to_application(self, raw: dict) -> Application:
        return Application(application_id=raw["ext_app"], candidate_id=raw["ext_cand"], job_id=raw["ext_req"])
```

- [ ] **Step 3: Implement `outbound.py`:**
```python
from __future__ import annotations
from typing import Callable, List, Optional, TypeVar
from ..data.audit import AuditStore

T = TypeVar("T")

class OutboundBlocked(Exception):
    pass

class OutboundGuard:
    def __init__(self, *, fully_local: bool = True, audit: Optional[AuditStore] = None,
                 actor: str = "system") -> None:
        self.fully_local = fully_local
        self._audit = audit
        self._actor = actor
    def send(self, *, target: str, fields: List[str], reason: str, deid_status: str = "none",
             call: Callable[[], T]) -> T:
        detail = f"{reason} fields=[{','.join(fields)}] deid={deid_status}"
        if self.fully_local:
            if self._audit is not None:
                self._audit.record(self._actor, "egress", target, reason=detail, result="rejected:fully_local")
            raise OutboundBlocked(f"fully-local mode: outbound to {target} blocked")
        if self._audit is not None:
            self._audit.record(self._actor, "egress", target, reason=detail, result="ok")
        return call()
```

```python
from __future__ import annotations
from dataclasses import dataclass
from .outbound import OutboundGuard
from .sdk import AntiCorruptionLayer, Connector
from ..data.store import CanonicalStore

@dataclass(frozen=True)
class IngestResult:
    kind: str
    count: int

class IntegrationService:
    def __init__(self, store: CanonicalStore, guard: OutboundGuard) -> None:
        self._store = store
        self._guard = guard
    def ingest(self, connector: Connector, acl: AntiCorruptionLayer, kind: str) -> IngestResult:
        records = self._guard.send(target=connector.name, fields=[kind], reason=f"pull:{kind}",
                                   call=lambda: connector.fetch(kind))
        upsert = {"candidate": self._store.upsert_candidate, "job": self._store.upsert_job,
                  "application": self._store.upsert_application}[kind]
        n = 0
        for rec in records:
            upsert(acl.translate(rec))
            n += 1
        return IngestResult(kind=kind, count=n)
```

```python
from __future__ import annotations
from typing import List
from ..core.tools import ToolRegistry, ToolSpec
from .outbound import OutboundBlocked
from .sdk import AntiCorruptionLayer, Connector
from .service import IntegrationService

def connector_toolspecs(service: IntegrationService, connector: Connector,
                        acl: AntiCorruptionLayer, kinds: List[str]) -> List[ToolSpec]:
    specs: List[ToolSpec] = []
    for kind in kinds:
        def make(kind: str):
            def handler(_args: dict) -> str:
                try:
                    res = service.ingest(connector, acl, kind)
                    return f"Ingested {res.count} {kind}(s) from {connector.name}."
                except OutboundBlocked:
                    return f"blocked: fully-local mode is on; {connector.name} egress is disabled."
            return handler
        specs.append(ToolSpec(
            name=f"{connector.name}_pull_{kind}",
            description=f"Pull {kind} records from the {connector.name} connector into the local store (outbound; governed by the fully-local switch).",
            parameters={"type": "object", "properties": {}},
            handler=make(kind),
        ))
    return specs

def register_connector(registry: ToolRegistry, *toolspecs: ToolSpec) -> None:
    for spec in toolspecs:
        registry.register(spec)
```
Note: in Step 1 use `specs[0].handler({})` directly (the test's `execute_by_name` branch is a fallback only if such a method exists — verify `ToolRegistry`'s execute API at run time and simplify the test to call the handler if needed).
