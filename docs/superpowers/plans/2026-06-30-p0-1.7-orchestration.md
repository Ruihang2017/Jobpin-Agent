# §1.7 Layer B Orchestration Skeleton — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:executing-plans (inline, single-agent). Steps
> use checkbox (`- [ ]`) syntax.

**Goal:** An in-house lightweight business-process state machine (Layer B) meeting three persistence
contracts — crash recovery, cross-day pause/resume, external side-effect idempotency — proven with a toy
process. Net-new; no Hermes port; no `agent_loop.py` change.

**Architecture:** New `orchestration/` package: a SQLite store (instances + append-only transitions +
idempotency), a declarative `ProcessDefinition` + `ProcessEngine` (validated transitions, persisted +
logged), `run_once` idempotency, a `recover` loader. Spec:
`docs/superpowers/specs/2026-06-30-p0-1.7-orchestration-design.md`.

**Tech Stack:** Python ≥3.11, stdlib only (`sqlite3`, `datetime`, `dataclasses`, `enum`), pytest.

## Global Constraints
- **Bilingual docstrings (EN then 中文)** on every file/class/function incl. tests (agent/CLAUDE.md); never
  a literal triple-quote inside a docstring.
- **Net-new** — no Hermes port, no `THIRD_PARTY_NOTICES` entry.
- **No `agent_loop.py` change** — the engine is standalone Layer B.
- Durability tests that simulate a "restart" MUST use a **file** `db_path` (a fresh store over the same

- Inject a `clock` into the engine for deterministic timestamps where useful (default = UTC now).

```python
# orchestration/store.py
from __future__ import annotations
import sqlite3
from pathlib import Path
from typing import List, Optional
from .state_machine import ProcessInstance, Status, Transition

class OrchestrationStore:
    def __init__(self, db_path: str = ":memory:") -> None:
        if db_path != ":memory:":
            Path(db_path).parent.mkdir(parents=True, exist_ok=True)
        self._conn = sqlite3.connect(db_path)
        self._conn.execute("CREATE TABLE IF NOT EXISTS process_instances ("
            "instance_id TEXT PRIMARY KEY, process_type TEXT, current_state TEXT, status TEXT, "
            "context_ref TEXT, updated_at TEXT)")
        self._conn.execute("CREATE TABLE IF NOT EXISTS transitions ("
            "id INTEGER PRIMARY KEY AUTOINCREMENT, instance_id TEXT, from_state TEXT, to_state TEXT, "
            "trigger TEXT, at TEXT, actor TEXT)")
        self._conn.execute("CREATE TABLE IF NOT EXISTS idempotency ("
            "key TEXT PRIMARY KEY, status TEXT, result TEXT, at TEXT)")
        self._conn.commit()

    def save_instance(self, inst: ProcessInstance) -> None:
        self._conn.execute("INSERT OR REPLACE INTO process_instances VALUES (?, ?, ?, ?, ?, ?)",
            (inst.instance_id, inst.process_type, inst.current_state, inst.status.value,
             inst.context_ref, inst.updated_at))
        self._conn.commit()

    def load_instance(self, instance_id: str) -> Optional[ProcessInstance]:
        r = self._conn.execute("SELECT instance_id, process_type, current_state, status, context_ref, "
            "updated_at FROM process_instances WHERE instance_id=?", (instance_id,)).fetchone()
        if r is None:
            return None
        return ProcessInstance(r[0], r[1], r[2], Status(r[3]), context_ref=r[4], updated_at=r[5])

    def append_transition(self, t: Transition) -> None:
        self._conn.execute("INSERT INTO transitions (instance_id, from_state, to_state, trigger, at, actor) "
            "VALUES (?, ?, ?, ?, ?, ?)", (t.instance_id, t.from_state, t.to_state, t.trigger, t.at, t.actor))
        self._conn.commit()

    def transitions_for(self, instance_id: str) -> List[Transition]:
        rows = self._conn.execute("SELECT instance_id, from_state, to_state, trigger, at, actor FROM "
            "transitions WHERE instance_id=? ORDER BY id", (instance_id,)).fetchall()
        return [Transition(*row) for row in rows]

    def non_terminal_instances(self) -> List[ProcessInstance]:
        rows = self._conn.execute("SELECT instance_id FROM process_instances WHERE status IN (?, ?, ?)",
            (Status.RUNNING.value, Status.SUSPENDED.value, Status.AWAITING_HITL.value)).fetchall()
        return [self.load_instance(r[0]) for r in rows]

    def idem_get(self, key: str) -> Optional[dict]:
        r = self._conn.execute("SELECT status, result FROM idempotency WHERE key=?", (key,)).fetchone()
        return None if r is None else {"status": r[0], "result": r[1]}

    def idem_put(self, key: str, status: str, result: str, at: str = "") -> None:
        self._conn.execute("INSERT OR REPLACE INTO idempotency VALUES (?, ?, ?, ?)", (key, status, result, at))
        self._conn.commit()
```

```python
from __future__ import annotations
from dataclasses import dataclass, field
from datetime import datetime, timezone
from enum import Enum
from typing import Callable, Dict, Set

class Status(str, Enum):
    RUNNING = "running"; SUSPENDED = "suspended"; AWAITING_HITL = "awaiting_hitl"
    DONE = "done"; FAILED = "failed"

class IllegalTransition(Exception): ...

@dataclass
class ProcessInstance:
    instance_id: str; process_type: str; current_state: str; status: Status
    context_ref: str = ""; updated_at: str = ""

@dataclass
class Transition:
    instance_id: str; from_state: str; to_state: str; trigger: str; at: str; actor: str

@dataclass
class ProcessDefinition:
    process_type: str; initial_state: str
    transitions: Dict[str, Set[str]]
    hitl_states: Set[str] = field(default_factory=set)
    suspend_states: Set[str] = field(default_factory=set)
    terminal_states: Set[str] = field(default_factory=set)
    def is_legal(self, frm: str, to: str) -> bool:
        return to in self.transitions.get(frm, set())
    def status_for(self, state: str) -> Status:
        if state in self.terminal_states: return Status.DONE
        if state in self.hitl_states: return Status.AWAITING_HITL
        if state in self.suspend_states: return Status.SUSPENDED
        return Status.RUNNING

def _utcnow() -> str:
    return datetime.now(timezone.utc).isoformat()

class ProcessEngine:
    def __init__(self, store, definition: ProcessDefinition, *, clock: Callable[[], str] = _utcnow):
        self.store = store; self.definition = definition; self._clock = clock

    def start(self, instance_id, *, context_ref="", actor="system") -> ProcessInstance:
        d = self.definition; now = self._clock()
        inst = ProcessInstance(instance_id, d.process_type, d.initial_state, Status.RUNNING,
                               context_ref=context_ref, updated_at=now)
        self.store.save_instance(inst)
        self.store.append_transition(Transition(instance_id, "", d.initial_state, "start", now, actor))
        return inst

    def transition(self, instance_id, to_state, *, trigger, actor="system",
                   force_status: Status | None = None) -> ProcessInstance:
        inst = self.store.load_instance(instance_id)
        if inst is None:
            raise IllegalTransition(f"unknown instance {instance_id!r}")
        if not self.definition.is_legal(inst.current_state, to_state):
            raise IllegalTransition(f"{inst.current_state!r} -> {to_state!r} not declared")
        now = self._clock(); frm = inst.current_state
        inst.current_state = to_state
        inst.status = force_status or self.definition.status_for(to_state)
        inst.updated_at = now
        self.store.save_instance(inst)
        self.store.append_transition(Transition(instance_id, frm, to_state, trigger, now, actor))
        return inst

    def await_hitl(self, instance_id, *, to_state, trigger, actor="system"):
        return self.transition(instance_id, to_state, trigger=trigger, actor=actor,
                               force_status=Status.AWAITING_HITL)
    def resume_hitl(self, instance_id, *, to_state, decision, actor="system"):
        return self.transition(instance_id, to_state, trigger=f"hitl:{decision}", actor=actor)
    def suspend(self, instance_id, *, to_state, trigger, actor="system"):
        return self.transition(instance_id, to_state, trigger=trigger, actor=actor,
                               force_status=Status.SUSPENDED)
    def resume(self, instance_id, *, to_state, trigger, actor="system"):
        return self.transition(instance_id, to_state, trigger=trigger, actor=actor)
    def fail(self, instance_id, *, reason, actor="system"):
        inst = self.store.load_instance(instance_id); now = self._clock(); frm = inst.current_state
        inst.status = Status.FAILED; inst.updated_at = now; self.store.save_instance(inst)
        self.store.append_transition(Transition(instance_id, frm, frm, f"fail:{reason}", now, actor))
        return inst
```
*(Execution note: `await_hitl`/`suspend` set the status explicitly via `force_status` because the
to_state's class already maps there; `resume_hitl`/`resume` go back to a RUNNING/terminal state via the
normal `status_for`. The hitl/suspend states must be declared in `transitions` both ways.)*

```python
# orchestration/idempotency.py
from __future__ import annotations
from datetime import datetime, timezone
from typing import Callable, Tuple
from .store import OrchestrationStore

class IdempotencyStore:
    def __init__(self, store: OrchestrationStore) -> None:
        self._store = store
    def run_once(self, key: str, effect_fn: Callable[[], str]) -> Tuple[str, bool]:
        existing = self._store.idem_get(key)
        if existing is not None:
            return existing.get("result", ""), False          # dedup-wins: present key -> skip, never re-run
        now = datetime.now(timezone.utc).isoformat()
        self._store.idem_put(key, "pending", "", now)          # register intent BEFORE executing
        result = effect_fn()                                   # the external side effect
        self._store.idem_put(key, "done", str(result), now)    # mark done + record the result
        return str(result), True
```

```python
from jobpin_agent.orchestration.idempotency import IdempotencyStore
from jobpin_agent.orchestration.recovery import recover
from jobpin_agent.orchestration.state_machine import ProcessDefinition, ProcessEngine, Status
from jobpin_agent.orchestration.store import OrchestrationStore

def _defn():
    return ProcessDefinition("hiring", "new",
        transitions={"new": {"screening"}, "screening": {"awaiting_review"},
                     "awaiting_review": {"offer"}, "offer": set()},
        hitl_states={"awaiting_review"}, terminal_states={"offer"})

def test_contract1_crash_recovery(tmp_path):
    db = str(tmp_path / "orch.db")
    e = ProcessEngine(OrchestrationStore(db), _defn())
    e.start("i1", context_ref="cand:7f3a")
    e.transition("i1", "screening", trigger="advance")
    e.await_hitl("i1", to_state="awaiting_review", trigger="need review", actor="alice")
    del e                                              # "kill" the process
    e2 = ProcessEngine(OrchestrationStore(db), _defn())  # fresh engine over the SAME file
    resumed = recover(e2.store)
    assert [(i.instance_id, i.status) for i in resumed] == [("i1", Status.AWAITING_HITL)]
    assert e2.store.load_instance("i1").context_ref == "cand:7f3a"   # context intact
    e2.resume_hitl("i1", to_state="offer", decision="approve", actor="bob")
    assert e2.store.load_instance("i1").status == Status.DONE        # resumes, no restart-from-zero

def test_contract2_cross_day_pause_resume(tmp_path):
    db = str(tmp_path / "orch.db")
    defn = ProcessDefinition("hiring", "new",
        transitions={"new": {"awaiting_event"}, "awaiting_event": {"done"}, "done": set()},
        suspend_states={"awaiting_event"}, terminal_states={"done"})
    e = ProcessEngine(OrchestrationStore(db), defn)
    e.start("i1", context_ref="cand:x")
    e.suspend("i1", to_state="awaiting_event", trigger="await external event")
    assert e.store.load_instance("i1").status == Status.SUSPENDED
    # ... arbitrarily long later (no wall-clock assumption); the event arrives, over a fresh engine:
    e2 = ProcessEngine(OrchestrationStore(db), defn)
    e2.resume("i1", to_state="done", trigger="event arrived")
    assert e2.store.load_instance("i1").status == Status.DONE
    assert e2.store.load_instance("i1").context_ref == "cand:x"      # context intact across resume

def test_contract3_side_effect_idempotency(tmp_path):
    db = str(tmp_path / "orch.db")
    calls = []

    key = "email:req_812:cand_7f3a:offer"

    IdempotencyStore(OrchestrationStore(db)).run_once(key, send_email)  # retry after "restart"
    assert len(calls) == 1                              # the email is sent exactly once
```
